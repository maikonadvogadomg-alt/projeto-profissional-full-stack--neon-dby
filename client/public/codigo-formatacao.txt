// Detecta linhas de assinatura
const isSignatureLine = (line) => {
  const t = line.trim();
  return /^(Advogad[oa]|OAB|Termos em que|Nestes termos|Pede deferimento|Respeitosamente|Data:|Local e data|___)/i.test(t)
    || /^[A-Z\s]+,\s*\d{1,2}\s+de\s+\w+\s+de\s+\d{4}/i.test(t)
    || /OAB\s*\/?\s*[A-Z]{2}\s*\d+/i.test(t);
};

// Detecta linhas de citaГғВ§ГғВЈo
const isCitationLine = (line) => {
  const t = line.trim();
  return (t.startsWith('"') || t.startsWith('\u201c') || t.startsWith('\u00ab'))
    || /^(Art\.|Artigo|ГӮВ§|Sumula|Ementa:|EMENTA:)/i.test(t);
};

// Detecta linhas de cabeГғВ§alho
const isHeaderLine = (line) => {
  const t = line.trim();
  return /^(EXCELENT|MERITISSIM|ILUSTRISSIM|AO\s+JUIZ|AO\s+DESEMBARGADOR|SENHOR\s+JUIZ|SENHOR\s+DESEMBARGADOR|EGRE|TRIBUNAL|COMARCA|VARA\s|JUIZO|PROCESSO\s*(N|n)|AUTOS\s*(N|n)|AGRAVANTE|AGRAVAD[OA]|RECORRENTE|RECORRIDO|REQUERENTE|REQUERIDO|IMPETRANTE|IMPETRADO|EMBARGANTE|EMBARGADO|AUTOR[A]?:|RE[UГғВҡ]:|APELANTE|APELAD[OA])/i.test(t)
    || /^(AGRAVO|RECURSO|APELA[CГғВҮ]|MANDADO|PETI[CГғВҮ]|EMBARGOS|RECLAMA[CГғВҮ]|HABEAS|A[CГғВҮ][AГғВғ]O)/i.test(t)
    || /^(CPF|CNPJ|RG|OAB)[\s:]/i.test(t)
    || /^Advogad[oa].*OAB/i.test(t);
};

const formatMarkdown = (text) => {
  const s = formatSettings; // { fontFamily, fontSize, lineHeight, textAlign, paragraphIndent, citationIndent }
  const baseFont = 'font-family: ' + s.fontFamily + '; font-size: ' + s.fontSize + 'pt; line-height: ' + s.lineHeight;
  const noIndent = baseFont + '; text-align: ' + s.textAlign;
  const bodyIndent = baseFont + '; text-align: ' + s.textAlign + '; text-indent: ' + s.paragraphIndent + 'cm';
  const citStyle = baseFont + '; font-size: ' + Math.max(s.fontSize - 2, 8) + 'pt; line-height: 1; text-align: ' + s.textAlign + '; margin-left: ' + s.citationIndent + 'cm';
  const sigStyle = baseFont + '; text-align: center';
  const headerStyle = baseFont + '; text-align: center; font-weight: bold';
  const headerSubStyle = baseFont + '; text-align: center';
  const titleStyle = baseFont + '; text-align: ' + s.textAlign + '; font-weight: bold';

  // Divide em parГғВЎgrafos (linhas vazias separam)
  const lines = text.split('\n');
  const paragraphs = [];
  let current = [];
  for (const line of lines) {
    if (line.trim() === '') {
      if (current.length > 0) { paragraphs.push(current.join('<br />')); current = []; }
    } else { current.push(line); }
  }
  if (current.length > 0) paragraphs.push(current.join('<br />'));

  let headerDone = false;

  return paragraphs.map((p, idx) => {
    const plain = p.replace(/<br\s*\/?>/g, ' ').trim();

    // 1. Assinatura ГўВҶВ’ centralizado
    if (isSignatureLine(plain)) return '<p style="' + sigStyle + '">' + p + '</p>';

    // 2. CitaГғВ§ГғВЈo ГўВҶВ’ recuo esquerda, fonte menor
    if (isCitationLine(plain)) return '<p style="' + citStyle + '">' + p + '</p>';

    // 3. CabeГғВ§alho (primeiros 8 parГғВЎgrafos) ГўВҶВ’ centralizado
    if (!headerDone && idx < 8 && isHeaderLine(plain)) {
      const isMain = /^(EXCELENT|MERITISSIM|ILUSTRISSIM)/i.test(plain.trim());
      return '<p style="' + (isMain ? headerStyle : headerSubStyle) + '">' + p + '</p>';
    }
    if (!headerDone && idx > 0 && !isHeaderLine(plain)) headerDone = true;

    // 4. TГғВӯtulo (MAIГғВҡSCULAS ou I. II. 1.) ГўВҶВ’ negrito, SEM recuo
    const isAllCaps = plain === plain.toUpperCase() && plain.length > 3 && plain.length < 200;
    const isSectionTitle = /^[IVXLCDM]+[\.\)]\s/.test(plain.trim()) || /^\d+[\.\)]\s/.test(plain.trim());
    if (isAllCaps || isSectionTitle) return '<p style="' + titleStyle + '"><strong>' + p + '</strong></p>';

    // 5. Corpo de texto (longo ou com pontuaГғВ§ГғВЈo) ГўВҶВ’ COM recuo
    const isBodyText = plain.length > 80 || /[.;:,]\s/.test(plain);
    const style = (isBodyText && s.paragraphIndent > 0) ? bodyIndent : noIndent;
    return '<p style="' + style + '">' + p + '</p>';
  }).join('');
};
